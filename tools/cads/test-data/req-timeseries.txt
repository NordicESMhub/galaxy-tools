request = {
    "sky_type": "clear",
    "location": {"longitude": 11, "latitude": 60},
    "altitude": ["-999"],
    "date": ["2026-03-04/2026-03-04"],
    "time_step": "15minute",
    "time_reference": "universal_time",
    "data_format": "netcdf"
}

client = cdsapi.Client()
